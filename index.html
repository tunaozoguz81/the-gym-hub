const CACHE = 'gymhub-v5'; const ASSETS = [ '/the-gym-hub/', '/the-gym-hub/index.html', '/the-gym-hub/manifest.json', ]; self.addEventListener('install', e => { e.waitUntil( caches.open(CACHE).then(c => c.addAll(ASSETS)).then(() => self.skipWaiting()) ); }); self.addEventListener('activate', e => { e.waitUntil( caches.keys().then(keys => Promise.all(keys.filter(k => k !== CACHE).map(k => caches.delete(k))) ).then(() => self.clients.claim()) ); }); self.addEventListener('fetch', e => { e.respondWith( caches.match(e.request).then(cached => cached || fetch(e.request)) ); });
